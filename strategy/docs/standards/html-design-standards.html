HTML Design Standards

Visual consistency for all Adrata strategy documents. Copy this CSS as the foundation for any new HTML file.

Typography

Font Family

Always use MiSans as the primary font. Include these CDN links in every document:

<link rel="stylesheet" href="https://cdn.jsdelivr.net/npm/misans@4.0.0/lib/Normal/MiSans-Regular.min.css">
<link rel="stylesheet" href="https://cdn.jsdelivr.net/npm/misans@4.0.0/lib/Normal/MiSans-Medium.min.css">
<link rel="stylesheet" href="https://cdn.jsdelivr.net/npm/misans@4.0.0/lib/Normal/MiSans-Semibold.min.css">
<link rel="stylesheet" href="https://cdn.jsdelivr.net/npm/misans@4.0.0/lib/Normal/MiSans-Bold.min.css">

Colors

CSS Variables (Required)

:root {
    --bg: #fff;
    --text: #1a1a1a;
    --text-secondary: #333;
    --text-muted: #666;
    --border: #e5e5e5;
    --card-bg: #fafafa;
    --accent: #e65100;
    --highlight-bg: #fff5f0;
}

[data-theme="dark"] {
    --bg: #0a0a0a;
    --text: #e5e5e5;
    --text-secondary: #ccc;
    --text-muted: #999;
    --border: #333;
    --card-bg: #1a1a1a;
    --highlight-bg: #1a1008;
}

Accent Color

#e65100 — Use for links, buttons, highlights, borders on important boxes. This is the only accent color.

What NOT to Use

  • No green, blue, purple, or other accent colors
  • No hardcoded grays—always use CSS variables
  • No emojis in strategy documents

Layout

Container

body {
    max-width: 800px;  /* or 900px for dense content */
    margin: 0 auto;
    padding: 48px 24px;
}

Spacing

  • h2 sections: 48px top margin, 24px top padding, border-top
  • Paragraphs: 14-16px margin-bottom
  • Cards/boxes: 20-24px padding

Components

Theme Toggle (Required)

<button class="theme-toggle" onclick="toggleTheme()">
    <svg class="sun">...</svg>
    <svg class="moon">...</svg>
</button>

<script>
function toggleTheme() { 
    const html = document.documentElement; 
    const isDark = html.getAttribute('data-theme') === 'dark'; 
    html.setAttribute('data-theme', isDark ? 'light' : 'dark'); 
    localStorage.setItem('theme', isDark ? 'light' : 'dark'); 
}
document.addEventListener('DOMContentLoaded', () => { 
    const theme = localStorage.getItem('theme') || 'dark'; 
    document.documentElement.setAttribute('data-theme', theme); 
});
</script>

Info Box

.box { 
    background: var(--card-bg); 
    border: 1px solid var(--border); 
    border-radius: 8px; 
    padding: 20px; 
    margin: 20px 0; 
}
.box.accent { border-left: 4px solid var(--accent); }

Breadcrumb

.breadcrumb { font-size: 13px; margin-bottom: 16px; }
.breadcrumb a { color: var(--accent); text-decoration: none; font-weight: 600; }

Default Theme

Always default to dark mode. Set data-theme="dark" on the html element.

Favicon

<link rel="icon" type="image/x-icon" href="../../favicon.svg">

Adjust path based on document location in folder structure.

Template

Copy this boilerplate for any new document:

<!DOCTYPE html>
<html lang="en" data-theme="dark">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Document Title | Adrata</title>
    <link rel="icon" type="image/x-icon" href="../../favicon.svg">
    <link rel="stylesheet" href="https://cdn.jsdelivr.net/npm/misans@4.0.0/lib/Normal/MiSans-Regular.min.css">
    <link rel="stylesheet" href="https://cdn.jsdelivr.net/npm/misans@4.0.0/lib/Normal/MiSans-Medium.min.css">
    <link rel="stylesheet" href="https://cdn.jsdelivr.net/npm/misans@4.0.0/lib/Normal/MiSans-Semibold.min.css">
    <link rel="stylesheet" href="https://cdn.jsdelivr.net/npm/misans@4.0.0/lib/Normal/MiSans-Bold.min.css">
    <style>
        :root { --bg: #fff; --text: #1a1a1a; --text-secondary: #333; --text-muted: #666; --border: #e5e5e5; --card-bg: #fafafa; --accent: #e65100; }
        [data-theme="dark"] { --bg: #0a0a0a; --text: #e5e5e5; --text-secondary: #ccc; --text-muted: #999; --border: #333; --card-bg: #1a1a1a; }
        * { margin: 0; padding: 0; box-sizing: border-box; }
        body { font-family: 'MiSans', -apple-system, sans-serif; background: var(--bg); color: var(--text); line-height: 1.7; font-size: 16px; padding: 48px 24px; max-width: 800px; margin: 0 auto; }
        h1 { font-size: 42px; font-weight: 800; margin-bottom: 12px; }
        .subtitle { font-size: 17px; color: var(--text-muted); margin-bottom: 32px; }
        h2 { font-size: 24px; font-weight: 700; margin: 48px 0 16px; padding-top: 24px; border-top: 1px solid var(--border); }
        p { margin-bottom: 14px; color: var(--text-secondary); }
    </style>
</head>
<body>
    <h1>Document Title</h1>
    <div class="subtitle">Subtitle here</div>
    
    <!-- Content -->
    
    <script>
        function toggleTheme() { 
            const html = document.documentElement; 
            const isDark = html.getAttribute('data-theme') === 'dark'; 
            html.setAttribute('data-theme', isDark ? 'light' : 'dark'); 
            localStorage.setItem('theme', isDark ? 'light' : 'dark'); 
        }
        document.addEventListener('DOMContentLoaded', () => { 
            const theme = localStorage.getItem('theme') || 'dark'; 
            document.documentElement.setAttribute('data-theme', theme); 
        });
    </script>
</body>
</html>
